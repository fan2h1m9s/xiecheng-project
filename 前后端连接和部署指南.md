# 前后端连接和数据库部署指南

## 📱 前端连接方案

### 1. 前端环境配置

在你的前端项目中创建 `.env` 文件：

```env
# .env.development
REACT_APP_API_BASE_URL=http://localhost:3000/api

# .env.production
REACT_APP_API_BASE_URL=https://api.yourdomain.com/api
```

### 2. 前端 API 调用示例

**使用 axios：**

```typescript
// api/client.ts
import axios from 'axios';

const apiClient = axios.create({
  baseURL: process.env.REACT_APP_API_BASE_URL,
  timeout: 10000,
});

// 添加请求拦截器 - 自动附加 JWT Token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('authToken');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// 添加响应拦截器 - 处理 401 未授权
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('authToken');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

export default apiClient;
```

### 3. React 组件中调用 API 示例

```typescript
// 用户登录
const login = async (userAccount: string, userPassword: string) => {
  try {
    const res = await apiClient.post('/users/login', {
      userAccount,
      userPassword,
    });
    localStorage.setItem('authToken', res.data.token);
    localStorage.setItem('user', JSON.stringify(res.data.user));
    return res.data;
  } catch (error) {
    console.error('登录失败:', error);
    throw error;
  }
};

// 获取酒店列表
const getHotels = async (page: number = 1, pageSize: number = 20) => {
  try {
    const res = await apiClient.get('/hotels', {
      params: { page, pageSize },
    });
    return res.data;
  } catch (error) {
    console.error('获取酒店列表失败:', error);
    throw error;
  }
};
```

## 🗄️ 数据库部署方案

### 1. MySQL 数据库安装

#### Windows 环境：

**方法一：使用 MySQL 安装程序**
- 下载 MySQL Community Server：https://dev.mysql.com/downloads/mysql/
- 运行安装程序，选择默认配置
- 启动 MySQL 服务：
  ```powershell
  # 检查服务状态
  Get-Service | grep -i mysql
  
  # 启动服务
  Start-Service MySQL80
  ```

**方法二：使用 Docker（推荐）**
```bash
docker run --name mysql-hotel \
  -e MYSQL_ROOT_PASSWORD=123456 \
  -e MYSQL_DATABASE=hotel \
  -p 3306:3306 \
  -d mysql:8.0
```

### 2. 创建数据库和表

**使用 MySQL 客户端连接：**

```bash
mysql -h localhost -u root -p
# 输入密码：123456
```

**执行以下 SQL 语句：**

```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS hotel CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- 切换到新数据库
USE hotel;

-- 创建表（TypeORM 会根据 entities 自动创建）
-- 如需手动创建，参考以下结构：

-- 用户表
CREATE TABLE IF NOT EXISTS `user` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `userAccount` varchar(255) NOT NULL UNIQUE,
  `userPassword` varchar(255) NOT NULL,
  `userName` varchar(255),
  `userType` int DEFAULT 1,
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 地址表
CREATE TABLE IF NOT EXISTS `address` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `province` varchar(255),
  `city` varchar(255),
  `district` varchar(255),
  `detailAddress` varchar(255),
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 酒店表
CREATE TABLE IF NOT EXISTS `hotel` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `hotelName` varchar(255) NOT NULL,
  `region` varchar(100),
  `status` int DEFAULT 1,
  `hotelIcon` varchar(255),
  `addressId` int,
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`addressId`) REFERENCES `address`(`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 房间类型表
CREATE TABLE IF NOT EXISTS `room_type` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `roomTypeName` varchar(255),
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 房间表
CREATE TABLE IF NOT EXISTS `room` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `roomNum` varchar(255) NOT NULL,
  `roomTypeId` int,
  `hotelId` int,
  `roomPrice` decimal(10, 2),
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`roomTypeId`) REFERENCES `room_type`(`id`),
  FOREIGN KEY (`hotelId`) REFERENCES `hotel`(`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 订单表
CREATE TABLE IF NOT EXISTS `order` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `orderNum` varchar(255) UNIQUE,
  `userId` int NOT NULL,
  `orderStatus` int DEFAULT 0,
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  `updatedAt` timestamp DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (`userId`) REFERENCES `user`(`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 订单房间关联表
CREATE TABLE IF NOT EXISTS `order_room_relation` (
  `id` int PRIMARY KEY AUTO_INCREMENT,
  `orderId` int NOT NULL,
  `roomId` int NOT NULL,
  `checkInDate` date,
  `checkOutDate` date,
  `guestCount` int,
  `createdAt` timestamp DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (`orderId`) REFERENCES `order`(`id`),
  FOREIGN KEY (`roomId`) REFERENCES `room`(`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

## 🚀 部署步骤

### 本地开发环境

1. **启动数据库**
   ```bash
   # 如果使用 Docker
   docker start mysql-hotel
   
   # 或启动本地 MySQL 服务
   ```

2. **启动 Redis**
   ```bash
   # 如果使用 Docker
   docker run -d -p 6379:6379 redis:latest
   ```

3. **启动后端服务**
   ```bash
   npm run dev
   ```

4. **启动前端服务**
   ```bash
   npm start  # React
   # 或
   npm run dev  # Vue/Vite
   ```

### 生产环境部署

#### 云服务器部署（以 Linux 为例）

**1. 安装 Node.js 和 npm**
```bash
curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
sudo apt-get install -y nodejs
```

**2. 安装 MySQL 和 Redis**
```bash
sudo apt-get install -y mysql-server redis-server
```

**3. 克隆项目并安装依赖**
```bash
git clone https://github.com/your-repo/hotel-server.git
cd hotel-server
npm install
npm run build
```

**4. 配置环境变量（生产环境）**
```env
NODE_ENV=production
PORT=3000
DB_HOST=localhost
DB_USERNAME=hotel_user
DB_PASSWORD=strong_password_here
DB_DATABASE=hotel
REDIS_HOST=localhost
FRONTEND_URL=https://yourdomain.com
```

**5. 使用 PM2 管理进程**
```bash
npm install -g pm2
pm2 start dist/index.js --name "hotel-server"
pm2 save
pm2 startup
```

**6. Nginx 反向代理配置**
```nginx
server {
    listen 80;
    server_name api.yourdomain.com;
    
    location / {
        proxy_pass http://localhost:3000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
```

#### Docker Compose 一键部署

创建 `docker-compose.yml`：

```yaml
version: '3.8'

services:
  backend:
    build: .
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - DB_HOST=mysql
      - DB_USERNAME=root
      - DB_PASSWORD=123456
      - DB_DATABASE=hotel
      - REDIS_HOST=redis
    depends_on:
      - mysql
      - redis
    networks:
      - hotel-network

  mysql:
    image: mysql:8.0
    environment:
      - MYSQL_ROOT_PASSWORD=123456
      - MYSQL_DATABASE=hotel
    ports:
      - "3306:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    networks:
      - hotel-network

  redis:
    image: redis:latest
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - hotel-network

volumes:
  mysql_data:
  redis_data:

networks:
  hotel-network:
    driver: bridge
```

启动所有服务：
```bash
docker-compose up -d
```

## 🔗 API 调用流程总结

```
前端应用
  ↓
1. 用户登录 POST /api/users/login
  ↓
2. 保存 Token 到 localStorage
  ↓
3. 所有后续请求在 Authorization header 中附带 Token
  ↓
4. 后端验证 JWT，执行业务逻辑
  ↓
5. 返回数据给前端
```
